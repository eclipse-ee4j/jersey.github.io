159     /**
160      * Run the data through a chain of stages identified by the root stage.
161      *
162      * @param <DATA>    processed data type.
163      * @param data      data to be processed.
164      * @param rootStage root stage of the stage chain.
165      * @return processing result.
166      */
167     public static <DATA> DATA process(DATA data, Stage<DATA> rootStage) {
168         Stage.Continuation<DATA> continuation = Stage.Continuation.of(data, rootStage);
169         Stage<DATA> currentStage;
170         while ((currentStage = continuation.next()) != null) {
171             continuation = currentStage.apply(continuation.result());
172         }
173         return continuation.result();
174     }
175 
176     /**
177      * Run the data through a chain of stages identified by the root stage.
178      *
179      * If an inflector is found in the leaf stage, it's reference is set into the {@code inflectorRef}
180      * parameter.
181      *
182      * @param <DATA>       processed data type.
183      * @param data         data to be processed.
184      * @param rootStage    root stage of the stage chain.
185      * @param inflectorRef a mutable reference to an inflector.
186      * @return processing result.
187      */
188     public static <DATA, RESULT, T extends Inflector<DATA, RESULT>> DATA process(
189             DATA data,
190             Stage<DATA> rootStage,
191             Ref<T> inflectorRef) {
192 
193         Stage<DATA> lastStage = rootStage;
194         Stage.Continuation<DATA> continuation = Stage.Continuation.of(data, lastStage);
195         while (continuation.next() != null) {
196             lastStage = continuation.next();
197             continuation = lastStage.apply(continuation.result());
198         }
199 
200         inflectorRef.set(Stages.<DATA, RESULT, T>extractInflector(lastStage));
201 
202         return continuation.result();
203     }

205     private static class StageChainBuilder<DATA> implements Stage.Builder<DATA> {
206 
207         private final Deque<Function<DATA, DATA>> transformations = new LinkedList<Function<DATA, DATA>>();
208         private Stage<DATA> rootStage;
209         private ChainableStage<DATA> lastStage;
210 
211         private StageChainBuilder(Function<DATA, DATA> transformation) {
212             transformations.push(transformation);
213         }
214 
215         private StageChainBuilder(ChainableStage<DATA> rootStage) {
216             this.rootStage = rootStage;
217             this.lastStage = rootStage;
218         }
219 
220         @Override
221         public Stage.Builder<DATA> to(Function<DATA, DATA> transformation) {
222             transformations.push(transformation);
223             return this;
224         }
225 
226         @Override
227         public Stage.Builder<DATA> to(final ChainableStage<DATA> stage) {
228             addTailStage(stage);
229             lastStage = stage;
230 
231             return this;
232         }
233 
234         private void addTailStage(Stage<DATA> lastStage) {
235             Stage<DATA> tail = lastStage;
236             if (!transformations.isEmpty()) {
237                 tail = convertTransformations(lastStage);
238             }
239             if (rootStage != null) {
240                 this.lastStage.setDefaultNext(tail);
241             } else {
242                 rootStage = tail;
243             }
244         }
245 
246         @Override
247         public Stage<DATA> build(Stage<DATA> stage) {
248             addTailStage(stage);
249 
250             return rootStage;
251         }
252 
253         @Override
254         public Stage<DATA> build() {
255             return build(null);
256         }
257 
258         private Stage<DATA> convertTransformations(Stage<DATA> successor) {
259             Stage<DATA> stage;
260             if (successor == null) {
261                 stage = new LinkedStage<DATA>(transformations.poll());
262             } else {
263                 stage = new LinkedStage<DATA>(transformations.poll(), successor);
264             }
265 
266             Function<DATA, DATA> t;
267             while ((t = transformations.poll()) != null) {
268                 stage = new LinkedStage<DATA>(t, stage);
269             }
270 
271             return stage;
272         }
273     }

275     /**
276      * Linked linear stage implementation.
277      *
278      * @param <DATA> processed data type.
279      */
280     public static class LinkedStage<DATA> implements Stage<DATA> {
281 
282         private final Stage<DATA> nextStage;
283         private final Function<DATA, DATA> transformation;
284 
285         /**
286          * Create a new stage that will return the supplied stage in the
287          * continuation.
288          *
289          * @param transformation Request transformation function to be applied in the stage.
290          * @param nextStage      next stage returned in the continuation.
291          */
292         public LinkedStage(Function<DATA, DATA> transformation, Stage<DATA> nextStage) {
293             this.nextStage = nextStage;
294             this.transformation = transformation;
295         }
296 
297         /**
298          * Create a new terminal stage .
299          *
300          * @param transformation Request transformation function to be applied in the stage.
301          */
302         public LinkedStage(Function<DATA, DATA> transformation) {
303             this(transformation, null);
304         }
305 
306         @Override
307         public Stage.Continuation<DATA> apply(DATA data) {
308             return Continuation.of(transformation.apply(data), nextStage);
309         }
310     }
