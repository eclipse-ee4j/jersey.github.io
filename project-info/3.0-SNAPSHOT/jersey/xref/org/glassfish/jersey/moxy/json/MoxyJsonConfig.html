73      /**
74       * Create a new configuration for {@link org.eclipse.persistence.jaxb.rs.MOXyJsonProvider}. If the
75       * {@code initDefaultProperties} is set to {@code true} then the following values are set:
76       * <ul>
77       *     <li>{@link javax.xml.bind.Marshaller#JAXB_FORMATTED_OUTPUT} - {@code false}</li>
78       *     <li>{@link org.eclipse.persistence.jaxb.JAXBContextProperties#JSON_INCLUDE_ROOT} - {@code false}</li>
79       *     <li>{@link org.eclipse.persistence.jaxb.MarshallerProperties#JSON_MARSHAL_EMPTY_COLLECTIONS} - {@code true}</li>
80       *     <li>{@link org.eclipse.persistence.jaxb.JAXBContextProperties#JSON_NAMESPACE_SEPARATOR} -
81       *         {@link org.eclipse.persistence.oxm.XMLConstants#DOT}</li>
82       * </ul>
83       *
84       * @param initDefaultProperties flag to determine whether the default values of properties mentioned above should be set or
85       * not.
86       */
87      public MoxyJsonConfig(final boolean initDefaultProperties) {
88          if (initDefaultProperties) {
89              // javax.xml.bind.Marshaller.JAXB_FORMATTED_OUTPUT
90              setFormattedOutput(false);
91  
92              // org.eclipse.persistence.jaxb.JAXBContextProperties.JSON_INCLUDE_ROOT
93              setIncludeRoot(false);
94  
95              // org.eclipse.persistence.jaxb.MarshallerProperties.JSON_MARSHAL_EMPTY_COLLECTIONS
96              setMarshalEmptyCollections(true);
97  
98              // org.eclipse.persistence.jaxb.JAXBContextProperties.JSON_NAMESPACE_SEPARATOR
99              setNamespaceSeparator(XMLConstants.DOT);
100         }
101     }
